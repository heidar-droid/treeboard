## Task 1 — Backend: `git diff --numstat` helper with cache

**Files:**
- Create: `src/arboviz/git_diff_stat.py`

The helper must:
1. Run `git -C <root> diff --numstat -- <rel_path>` on a path relative to repo root.
2. Parse the single line: `<added>\t<removed>\t<filename>`. `--numstat` reports `-\t-\t<filename>` for binary files; treat that as `None`.
3. Cache by `(abs_path, mtime_ns)` so a repeated edit of the same file at the same mtime doesn't shell out twice.
4. Return `None` if git is unavailable, the file isn't in a repo, or the file is untracked. **Never raise.**

```python
# src/arboviz/git_diff_stat.py
"""Cached `git diff --numstat` helper for agent diff badges.

Public surface:
  - DiffStat(added: int, removed: int)
  - diff_stat(root, rel_path) -> DiffStat | None

Never raises. Returns None for any failure mode (no git, no repo, binary,
untracked, etc.). Cache key is (abs_path, mtime_ns) so the same file at the
same mtime never shells out twice — repeated `arboviz edit` events on the
same file between writes hit the cache.
"""
from __future__ import annotations

import os
import subprocess
from dataclasses import dataclass
from pathlib import Path
from typing import Optional


@dataclass(frozen=True)
class DiffStat:
    added: int
    removed: int


_CACHE: dict[tuple[str, int], Optional[DiffStat]] = {}
_CACHE_MAX = 512


def _mtime_ns(p: Path) -> int:
    try:
        return p.stat().st_mtime_ns
    except OSError:
        return 0


def diff_stat(root: Path | str, rel_path: str) -> Optional[DiffStat]:
    root_p = Path(root)
    abs_p = (root_p / rel_path).resolve()
    if not abs_p.is_file():
        return None
    key = (str(abs_p), _mtime_ns(abs_p))
    if key in _CACHE:
        return _CACHE[key]

    try:
        proc = subprocess.run(
            ["git", "-C", str(root_p), "diff", "--numstat", "--", rel_path],
            capture_output=True, text=True, timeout=2, check=False,
        )
    except (FileNotFoundError, subprocess.TimeoutExpired):
        result: Optional[DiffStat] = None
    else:
        result = _parse(proc.stdout) if proc.returncode == 0 else None

    if len(_CACHE) >= _CACHE_MAX:
        # Crude FIFO eviction — enough for a dev-tool session.
        try:
            _CACHE.pop(next(iter(_CACHE)))
        except StopIteration:
            pass
    _CACHE[key] = result
    return result


def _parse(stdout: str) -> Optional[DiffStat]:
    line = stdout.strip().split("\n", 1)[0].strip()
    if not line:
        return None
    parts = line.split("\t")
    if len(parts) < 3:
        return None
    a_str, r_str = parts[0], parts[1]
    # Binary files: git emits `-\t-\t<path>`
    if a_str == "-" or r_str == "-":
        return None
    try:
        return DiffStat(added=int(a_str), removed=int(r_str))
    except ValueError:
        return None


def _reset_cache_for_tests() -> None:
    _CACHE.clear()
```

```javascript
// src/arboviz/static/history-clock.js
//
// Bottom-left clock pill that shows the task count and reveals a popover
// listing past tasks on hover or `T` keypress. Replaces the v2.0 strip.
//
// Lifecycle:
//   - Idle (timeline empty): hidden
//   - First task-end: fade in + single 1.6s pulse (discoverability nudge)
//   - Hover or T: popover slides up
//   - Mouseout + 240ms delay OR Esc: popover collapses
//   - Click entry: agentState.viewPastTask(i) or viewLive() for last+live

export function setupHistoryClock(viewport, agentState) {
  if (getComputedStyle(viewport).position === "static") {
    viewport.style.position = "relative";
  }

  const pill = document.createElement("button");
  pill.id = "history-clock";
  pill.type = "button";
  pill.setAttribute("aria-label", "task history");
  pill.style.display = "none";
  pill.innerHTML = `
    <span class="glyph" aria-hidden="true"></span>
    <span class="lbl">history</span>
    <span class="count">0</span>
  `;
  viewport.appendChild(pill);

  const pop = document.createElement("div");
  pop.id = "history-popover";
  pop.setAttribute("role", "dialog");
  pop.style.display = "none";
  viewport.appendChild(pop);

  let firstShownPulseUsed = false;
  let closeTimer = null;

  function openPopover() {
    if (closeTimer) { clearTimeout(closeTimer); closeTimer = null; }
    pop.style.display = "block";
    requestAnimationFrame(() => pop.classList.add("open"));
  }
  function scheduleClose() {
    if (closeTimer) clearTimeout(closeTimer);
    closeTimer = setTimeout(() => {
      pop.classList.remove("open");
      setTimeout(() => { if (!pop.classList.contains("open")) pop.style.display = "none"; }, 280);
    }, 240);
  }
  function closeNow() {
    if (closeTimer) { clearTimeout(closeTimer); closeTimer = null; }
    pop.classList.remove("open");
    setTimeout(() => { if (!pop.classList.contains("open")) pop.style.display = "none"; }, 280);
  }

  pill.addEventListener("mouseenter", openPopover);
  pill.addEventListener("mouseleave", scheduleClose);
  pop.addEventListener("mouseenter", () => { if (closeTimer) { clearTimeout(closeTimer); closeTimer = null; } });
  pop.addEventListener("mouseleave", scheduleClose);

  window.addEventListener("keydown", (e) => {
    if (e.key === "Escape" && pop.classList.contains("open")) closeNow();
    // `T` toggles, but only when not inside an input/textarea
    const t = document.activeElement;
    if (e.key === "t" && (!t || !["INPUT", "TEXTAREA"].includes(t.tagName))) {
      if (pop.classList.contains("open")) closeNow(); else openPopover();
    }
  });

  function renderPopover(timeline, activeFootprint) {
    const isLive = activeFootprint === null;
    pop.replaceChildren();
    const header = document.createElement("div");
    header.className = "head";
    header.textContent = `history · ${timeline.length} task${timeline.length === 1 ? "" : "s"}`;
    pop.appendChild(header);

    timeline.forEach((entry, i) => {
```

```javascript
// src/arboviz/static/live-status.js
//
// Top-center pill showing what the agent is doing RIGHT NOW.
// Visible only while a task is in flight; vanishes 800ms after task-end.

export function setupLiveStatus(viewport, agentState) {
  if (getComputedStyle(viewport).position === "static") {
    viewport.style.position = "relative";
  }
  const pill = document.createElement("div");
  pill.id = "live-status";
  pill.style.display = "none";
  pill.innerHTML = `
    <span class="pulse" aria-hidden="true"></span>
    <span class="verb">idle</span>
    <span class="path"></span>
    <span class="age" aria-hidden="true"></span>
  `;
  viewport.appendChild(pill);

  let taskStartMs = 0;
  let lastEvent = null;       // { verb, file }
  let ageTimer = null;
  let hideTimer = null;
  let visible = false;

  const VERB = { read: "reading", edit: "editing", create: "creating", delete: "deleting" };

  function relPath(absPath, rootPath) {
    if (!absPath) return "";
    if (rootPath && absPath.startsWith(rootPath + "/")) return absPath.slice(rootPath.length + 1);
    return absPath;
  }

  function show(verb, file) {
    if (hideTimer) { clearTimeout(hideTimer); hideTimer = null; }
    if (!visible) {
      pill.style.display = "inline-flex";
      requestAnimationFrame(() => pill.classList.add("open"));
      visible = true;
      taskStartMs = Date.now();
      startAgeTicker();
    }
    pill.querySelector(".verb").textContent = verb;
    const root = window.__tb?.tree?.path || "";
    pill.querySelector(".path").textContent = relPath(file, root);
    lastEvent = { verb, file };
  }

  function hide() {
    pill.classList.remove("open");
    hideTimer = setTimeout(() => {
      if (!visible) return;
      pill.style.display = "none";
      visible = false;
      stopAgeTicker();
      pill.querySelector(".age").textContent = "";
    }, 800);
  }

  function startAgeTicker() {
    stopAgeTicker();
    const tick = () => {
      if (!visible) return;
      const s = Math.max(0, Math.round((Date.now() - taskStartMs) / 1000));
      pill.querySelector(".age").textContent = `${s}s`;
    };
    tick();
    ageTimer = setInterval(tick, 500);
  }
  function stopAgeTicker() {
    if (ageTimer) { clearInterval(ageTimer); ageTimer = null; }
  }

  let prevCanvasState = "idle";

  agentState.subscribe((s) => {
    const cs = s.canvasState;
    // task-end → schedule hide
    if (prevCanvasState !== "frozen" && cs === "frozen") {
      hide();
    }
    // any active op → show with the most recent event
    if (cs === "scanning" || cs === "editing") {
      // The latest op in agentOps wins. Fall back to "scanning" verb when
      // a snapshot just fired.
      let verb = "scanning";
      let file = "";
      if (s.agentOps.size > 0) {
        // Pick the most recently-set entry (Map preserves insertion order).
        const lastKey = [...s.agentOps.keys()].at(-1);
        const op = s.agentOps.get(lastKey);
        verb = VERB[op] || op;
        file = lastKey;
      }
      show(verb, file);
    }
    prevCanvasState = cs;
  });

  return { dispose: () => { stopAgeTicker(); pill.remove(); } };
}
```

```javascript
// src/arboviz/static/diff-badge.js
//
// Inline +N/-N badge anchored below each touched pill. Reads from
// agentState.agentDiffs. Re-positions on every render — agent-pills.js
// is the canonical pill-state driver, so we subscribe to the same source.

const BADGE_CLASS = "diff-badge";

export function setupDiffBadges(viewport, board, agentState) {
  if (getComputedStyle(viewport).position === "static") {
    viewport.style.position = "relative";
  }
  const layer = document.createElement("div");
  layer.id = "diff-badge-layer";
  layer.style.cssText = "position:absolute;inset:0;pointer-events:none;z-index:21;";
  viewport.appendChild(layer);

  function render() {
    layer.replaceChildren();
    const diffs = agentState.agentDiffs;
    if (!diffs || diffs.size === 0) return;
    const vpRect = viewport.getBoundingClientRect();
    for (const [path, stat] of diffs) {
      // Find the pill in the live DOM
      const node = board.querySelector(`g.node[data-path="${CSS.escape(path)}"]`);
      if (!node) continue;
      const rect = node.querySelector("rect.pill");
      if (!rect) continue;
      const r = rect.getBoundingClientRect();
      const x = r.left + r.width / 2 - vpRect.left;
      const y = r.bottom + 4 - vpRect.top;

      const badge = document.createElement("div");
      badge.className = BADGE_CLASS;
      badge.setAttribute("data-path", path);
      badge.style.left = `${x}px`;
      badge.style.top = `${y}px`;
      const parts = [];
      if (stat.added) parts.push(`<span class="plus">+${stat.added}</span>`);
      if (stat.removed) parts.push(`<span class="minus">−${stat.removed}</span>`);
      badge.innerHTML = parts.join("");
      layer.appendChild(badge);
    }
  }

  // Re-render on agent-state change AND on camera-driven redraws (the pills
  // move when the user pans/zooms or the file watcher triggers redraw).
  agentState.subscribe(render);

  // Pan/zoom: camera transforms the world; re-position on each frame the
  // transform changes. Cheap MutationObserver on board for size changes;
  // cheap rAF tick when world.style.transform changes.
  const world = document.getElementById("world");
  let rafToken = 0;
  function schedule() {
    if (rafToken) return;
    rafToken = requestAnimationFrame(() => { rafToken = 0; render(); });
  }
  new MutationObserver(schedule).observe(world, {
    attributes: true, attributeFilter: ["style", "transform"],
  });
  new ResizeObserver(schedule).observe(board);
  // Also re-render when the watcher rebuilds the tree (board mutates).
  new MutationObserver(schedule).observe(board, { childList: true, subtree: true });

  render();
  return { dispose: () => layer.remove() };
}
```
